### 2. PriceValidator（價格驗證器）
```python
# backend/app/validators/price_validator.py

class PriceValidator:
    """價格數據驗證器"""
    
    def __init__(self, max_daily_change: float = 0.05):
        self.max_daily_change = max_daily_change
        
    def validate_price(self, price: float) -> bool:
        """驗證價格合理性"""
        if price <= 0:
            return False
        return True
        
    def validate_timestamp(self, ts: datetime) -> bool:
        """驗證時間戳"""
        now = datetime.now()
        if ts > now:
            return False  # 未來時間
        if (now - ts).days > 365:
            return False  # 超過1年
        return True
        
    def validate_price_change(self, old_price: float, new_price: float) -> bool:
        """驗證價格變動是否在合理範圍"""
        if old_price <= 0:
            return False
        change = abs(new_price - old_price) / old_price
        return change <= self.max_daily_change
```

### 3. MarketValidator（市場數據驗證器）
```python
# backend/app/validators/market_validator.py

class MarketValidator:
    """市場數據驗證器"""
    
    def validate_dxy(self, value: float) -> bool:
        """DXY 範圍驗證（通常 90-110）"""
        return 80 <= value <= 120
        
    def validate_rate(self, rate: float) -> bool:
        """利率驗證（通常 0-20%）"""
        return 0 <= rate <= 25
        
    def validate_volume(self, volume: float) -> bool:
        """成交量驗證"""
        return volume >= 0
```
