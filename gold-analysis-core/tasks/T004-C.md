### 2. DataQualityReport（品質報告）
```python
# backend/app/reports/data_quality.py

from typing import Dict, List
from dataclasses import dataclass
from datetime import datetime

@dataclass
class QualityMetrics:
    total_records: int
    valid_records: int
    invalid_records: int
    missing_count: int
    outlier_count: int
    duplicate_count: int
    quality_score: float

class DataQualityReport:
    """數據品質報告生成器"""
    
    def __init__(self):
        self.metrics = {}
        
    def generate_report(self, data: List[dict], validator_results: Dict, cleaner_results: Dict) -> Dict:
        """生成品質報告"""
        total = len(data)
        valid = validator_results.get('valid_count', 0)
        missing = cleaner_results.get('missing_fixed', 0)
        outliers = cleaner_results.get('outliers_removed', 0)
        duplicates = cleaner_results.get('duplicates_removed', 0)
        
        # 計算品質分數
        quality_score = self._calculate_quality_score(
            total, valid, missing, outliers, duplicates
        )
        
        return {
            'timestamp': datetime.now().isoformat(),
            'metrics': {
                'total_records': total,
                'valid_records': valid,
                'invalid_records': total - valid,
                'missing_count': missing,
                'outlier_count': outliers,
                'duplicate_count': duplicates,
                'quality_score': quality_score
            },
            'grade': self._get_grade(quality_score)
        }
        
    def _calculate_quality_score(self, total: int, valid: int, missing: int, 
                                   outliers: int, duplicates: int) -> float:
        """計算品質分數（0-100）"""
        if total == 0:
            return 0.0
        validity_score = valid / total * 40
        completeness_score = (total - missing) / total * 30
        consistency_score = (total - duplicates) / total * 30
        return min(100, validity_score + completeness_score + consistency_score)
        
    def _get_grade(self, score: float) -> str:
        """根據分數給出等級"""
        if score >= 90:
            return 'A'
        elif score >= 80:
            return 'B'
        elif score >= 70:
            return 'C'
        elif score >= 60:
            return 'D'
        return 'F'
```
