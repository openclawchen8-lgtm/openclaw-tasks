```python
# backend/app/cleaners/outlier_detector.py

import numpy as np
from typing import List, Tuple

class OutlierDetector:
    """異常值檢測器"""
    
    def detect_zscore(self, data: List[float], threshold: float = 3.0) -> List[int]:
        """Z-score 方法檢測異常"""
        arr = np.array(data)
        mean = np.mean(arr)
        std = np.std(arr)
        if std == 0:
            return []
        z_scores = np.abs((arr - mean) / std)
        return [i for i, z in enumerate(z_scores) if z > threshold]
        
    def detect_iqr(self, data: List[float], k: float = 1.5) -> List[int]:
        """IQR 方法檢測異常"""
        arr = np.array(data)
        q1, q3 = np.percentile(arr, [25, 75])
        iqr = q3 - q1
        lower = q1 - k * iqr
        upper = q3 + k * iqr
        return [i for i, v in enumerate(arr) if v < lower or v > upper]
        
    def remove_outliers(self, data: List[dict], indices: List[int]) -> List[dict]:
        """移除異常值"""
        return [d for i, d in enumerate(data) if i not in indices]
```
