## 目標

滑鼠懸停在本地 `.md` 連結上 500ms，浮層顯示該 Markdown 檔案的渲染預覽內容。

## 實作方式

### 偵測連結懸停

```javascript
document.addEventListener('mouseover', function(e) {
    var target = e.target;
    while (target && target.tagName !== 'A') {
        target = target.parentNode;
    }
    if (!target) return;

    var href = target.getAttribute('href');
    if (!href || href.startsWith('http') || href.startsWith('mailto')) return;

    // 延遲 500ms
    window._hoverTimer = setTimeout(function() {
        previewLink(href, target);
    }, 500);
});

document.addEventListener('mouseout', function(e) {
    clearTimeout(window._hoverTimer);
    // 滑鼠離開連結時不立即關閉，等 300ms 緩衝（避免滑入預覽浮層又觸發 close）
    window._hoverCloseTimer = setTimeout(closePreview, 300);
});
```

### 解析相對路徑

```javascript
function previewLink(href, linkElement) {
    var baseDir = /* 從 currentFile 取得 */;
    var absPath = baseDir + '/' + href.replace(/^\//, '');

    // 透過 Go binding 讀取檔案
    if (window.previewMarkdownFile) {
        window.previewMarkdownFile(absPath, function(html) {
            showPreviewPopup(html, linkElement);
        });
    }
}
```

### 預覽浮層

```javascript
function showPreviewPopup(html, linkElement) {
    var popup = document.getElementById('linkPreviewPopup');
    if (!popup) {
        popup = document.createElement('div');
        popup.id = 'linkPreviewPopup';
        popup.style.cssText = 'position:absolute; z-index:99999; max-width:600px; max-height:400px; overflow:auto; background:var(--color-canvas-default); border:1px solid var(--color-border-default); border-radius:8px; box-shadow:0 4px 16px rgba(0,0,0,0.2); padding:12px; font-size:14px;';
        document.body.appendChild(popup);

        // 滑鼠進入浮層時取消 close 計時
        popup.addEventListener('mouseenter', function() {
            clearTimeout(window._hoverCloseTimer);
        });
        popup.addEventListener('mouseleave', function() {
            window._hoverCloseTimer = setTimeout(closePreview, 300);
        });
    }
    popup.innerHTML = '<div class="markdown-body">' + html + '</div>';
    popup.style.display = 'block';

    // 定位：放在連結下方或上方（視空間而定）
    var rect = linkElement.getBoundingClientRect();
    popup.style.left = rect.left + 'px';
    popup.style.top = (rect.bottom + window.scrollY + 4) + 'px';
}

function closePreview() {
    var popup = document.getElementById('linkPreviewPopup');
    if (popup) popup.style.display = 'none';
}
```

### Go binding

```go
wv.Bind("previewMarkdownFile", func(path string, callback string) {
    data, err := os.ReadFile(path)
    if err != nil {
        wv.Eval(callback + '("")')
        return
    }
    html, _ := renderer.Render(string(data))
    wv.Eval(callback + '("' + escapeJS(html) + '")')
})
```

### 限制

- 僅支援本地 `.md` 連結（相對路徑）
- 僅支援 `.md` / `.markdown` 副檔名
- 檔案不存在時不顯示浮層

## 驗收條件

- [ ] 滑鼠懸停本地 .md 連結 500ms 後顯示預覽浮層
- [ ] 浮層內容為該 Markdown 檔案的渲染結果
- [ ] 浮層正確顯示在連結附近（不超出視窗邊界）
- [ ] 滑鼠移入浮層時保持顯示，移出後 300ms 自動關閉
- [ ] 點擊連結正常跳轉（不受浮層影響）
